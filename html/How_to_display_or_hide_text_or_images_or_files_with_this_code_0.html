<script>
function unhide(spanID) {
                                                    var item = document.getElementById(spanID);
                                                         if (item) {
                                                              item.className=(item.className=='hidden')?'unhidden':'hidden';
                                                         }
                                                }
                                           
                                            function hide(spanID) {
                                                    var item = document.getElementById(spanID);
                                                         if (item) {
                                                            item.className='hidden';
                                                         }
                                                }
                                                function check(RadioID)
                                                {
                                                  document.getElementById(RadioID).checked=true;
                                                }
                                               
                                                function changeHiddenInput (objDropDown, hiddenID)
                                                {
                                                var objHidden = document.getElementById(hiddenID);
                                                var strUser = objDropDown.options[objDropDown.selectedIndex].value;
                                                objHidden.value = strUser;
                                                }
</script>
